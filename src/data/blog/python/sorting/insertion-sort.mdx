```python{numberLines: true}
def insertionSort(arr):
    
    for i in range(1, len(arr)):
        index = i

        while(index > 0):
            if(arr[index] < arr[index - 1]):
                tmp = arr[index - 1]
                arr[index - 1] = arr[index]
                arr[index] = tmp
                index-=1
            else:
                break

    return arr

```
